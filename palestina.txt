// Funci√≥n para desencriptar la IP (conexi√≥n) del jugador
function decryptHex(conn) {
  return conn.replace(/(..)/g, (_, a) => String.fromCharCode(parseInt(a, 16)));
}

// Funci√≥n para obtener la geolocalizaci√≥n de una IP
async function getGeoLocation(ip) {
  try {
    const res = await fetch(`https://ipapi.co/${ip}/json/`);
    const data = await res.json();
    return { country: data.country_name || "Desconocido" };
  } catch {
    return { country: "Desconocido" };
  }
}

// Funci√≥n para enviar mensajes al webhook de Discord
function sendDiscordLog(texto) {
  // Manejo de errores para el env√≠o al webhook
  fetch(LOG_DISCORD_WEBHOOK, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ content: texto })
  })
  .catch(error => console.error("Error al enviar log a Discord:", error)); 
}

    return;
  }

  // Inicializa los datos del jugador
  playerData[player.id] = { ip: ip, auth: player.auth, isMuted: false };
  
  room.sendAnnouncement(`üëã Bienvenido ${player.name}`);
  sendDiscordLog(`‚úÖ **ENTR√ì:** ${player.name} | Auth: \`${player.auth}\` | IP: \`${ip}\` | Pa√≠s: ${location.country}`);

  // Auto-admin por Auth
  if (player.auth === AUTO_ADMIN_AUTH) {

    room.sendAnnouncement(`üîë ${player.name} ha sido autom√°ticamente reconocido como admin.`, null, 0x00ff59, "bold", 2);
    sendDiscordLog(`üëë **AUTO-ADMIN:** ${player.name} recibi√≥ admin al unirse.`);
  }
};

      if (command === "!mute") {
        if (args.length < 2) {
          room.sendAnnouncement("Uso: !mute <ID_jugador>", player.id, 0xFF0000, "normal", 0);
          return false;
        }
        const targetPlayerId = parseInt(args[1]);
        const targetPlayer = room.getPlayerList().find(p => p.id === targetPlayerId);

        if (!targetPlayer) {
          room.sendAnnouncement("Jugador con esa ID no encontrado.", player.id, 0xFF0000, "normal", 0);
          return false;
        }
        if (isAdmin(targetPlayer)) {
          room.sendAnnouncement("No puedes mutear a otro admin.", player.id, 0xFF0000, "normal", 0);
          return false;
        }
        if (mutedPlayers.has(targetPlayer.id)) {
          room.sendAnnouncement(`${targetPlayer.name} ya est√° muteado.`, player.id, 0xFFA500, "normal", 0);
        } else {
          mutedPlayers.add(targetPlayer.id);
          playerData[targetPlayer.id].isMuted = true; // Actualizar el estado en playerData
          room.sendAnnouncement(`üîá ${targetPlayer.name} ha sido muteado por ${player.name}.`, null, 0xFF0000, "bold", 2);
        }
        return false;
      }

      // Comando: !unmute
      if (command === "!unmute") {
        if (args.length < 2) {
          room.sendAnnouncement("Uso: !unmute <ID_jugador>", player.id, 0xFF0000, "normal", 0);
          return false;
        }
        const targetPlayerId = parseInt(args[1]);
        const targetPlayer = room.getPlayerList().find(p => p.id === targetPlayerId);

        if (!targetPlayer) {
          room.sendAnnouncement("Jugador con esa ID no encontrado.", player.id, 0xFF0000, "normal", 0);
          return false;
        }
        if (!mutedPlayers.has(targetPlayer.id)) {
          room.sendAnnouncement(`${targetPlayer.name} no est√° muteado.`, player.id, 0xFFA500, "normal", 0);
        } else {
          mutedPlayers.delete(targetPlayer.id);
          playerData[targetPlayer.id].isMuted = false; // Actualizar el estado en playerData
          room.sendAnnouncement(`üîä ${targetPlayer.name} ha sido desmuteado por ${player.name}.`, null, 0x00FF00, "bold", 2);
        }
        return false;
      }

      // Comando: !mutelist
      if (command === "!mutelist") {
        if (mutedPlayers.size === 0) {
          room.sendAnnouncement("No hay jugadores muteados actualmente.", player.id, 0x00FF00, "normal", 0);
        } else {
          const mutedPlayerNames = Array.from(mutedPlayers).map(id => {
            const p = room.getPlayerList().find(player => player.id === id);
            return p ? `${p.name} (ID: ${p.id})` : `ID: ${id} (desconectado)`;
          }).join(", ");
          room.sendAnnouncement(`üîá Jugadores muteados: ${mutedPlayerNames}`, player.id, 0xFFA500, "normal", 0);
        }
        return false;
      }
      
      // Si es un comando que comienza con '!' pero no est√° reconocido (para admins)
      room.sendAnnouncement("Comando no encontrado.", player.id, 0xFF0000, "normal", 0);
      return false;

    } else { // Si NO es admin y el mensaje empieza con '!' (y no es !help o !federacionhaxballespa√±a)
      room.sendAnnouncement("Comando no reconocido o requiere permisos de administraci√≥n.", player.id, 0xFF0000, "normal", 0);
      return false;
    }
  }

  // --- Manejo del chat normal (no es un comando) ---
  let formattedMessage = "";
  let chatColor = 0xFFFFFF; // Blanco por defecto

  if (isAdmin(player)) {
    formattedMessage = `üëëADMINüëë - [${player.id}] - ${player.name}: ${message}`;
    chatColor = 0xFFFF00; // Amarillo para admins
  } else {
    formattedMessage = `[${player.id}] - ${player.name}: ${message}`;
  }

  room.sendAnnouncement(formattedMessage, null, chatColor, "bold", 0);
  return false; // Evitar que Haxball muestre el mensaje original
};

// Cuando se marca un gol
room.onTeamGoal = function (scoringTeam) {
  const players = room.getPlayerList();
  const lastTouch = [...touches].reverse().find(p => p.team !== 0); // √öltimo toque que no sea de espectador
  
  // Obtener el tiempo actual del juego en minutos y segundos
  const gameTime = room.getScores().time;
  const minutes = Math.floor(gameTime / 60);
  const seconds = Math.floor(gameTime % 60);
  const formattedTime = `${minutes}:${seconds < 10 ? '0' : ''}${seconds}`;

  let msg = "";
  let log = "";
  const teamName = scoringTeam === 1 ? "Rojo" : "Azul"; // Determina el nombre del equipo anotador

  if (!lastTouch) {
    msg = `‚ö†Ô∏è Gol no reconocido por el sistema (revisi√≥n manual sugerida) | Minuto: ${formattedTime}`;
    log = `‚ö†Ô∏è **GOL NO RECONOCIDO** | El sistema no detect√≥ ning√∫n toque v√°lido. Revisi√≥n manual recomendada. | Minuto: ${formattedTime}`;
    room.sendAnnouncement(msg, null, 0xffcc00, "bold", 2);
  } else {
    const player = players.find(p => p.id === lastTouch.id);

    if (!stats[player.id]) stats[player.id] = { goals: 0, assists: 0, ownGoals: 0 };

    if (lastTouch.team !== scoringTeam) {
      // Autogol
      stats[player.id].ownGoals++;
      msg = `${player.name}, creo que te confundiste de arco... auto-gol en el minuto ${formattedTime}`;
      log = `üíÄ **AUTOGOL** | ${player.name} (${lastTouch.team === 1 ? "Rojo" : "Azul"}), creo que te confundiste de arco... auto-gol en el minuto ${formattedTime}`;
      room.sendAnnouncement(msg, null, 0xFF0000, "bold", 2); // Rojo para autogoles
    } else {
      // Gol normal o con asistencia
      stats[player.id].goals++;
      const prevTouch = [...touches].reverse().find(p => p.team === scoringTeam && p.id !== lastTouch.id);
      
      if (prevTouch) {
        const asistidor = players.find(p => p.id === prevTouch.id);
        if (asistidor && asistidor.id !== player.id) { // Asegurarse de que el asistidor no sea el mismo goleador
          if (!stats[asistidor.id]) stats[asistidor.id] = { goals: 0, assists: 0, ownGoals: 0 };
          stats[asistidor.id].assists++;
          msg = `${player.name} (${teamName}) ha marcado gol con el pase de ${asistidor.name} en el minuto ${formattedTime}`;
          log = `‚öΩ **GOL + ASISTENCIA** | ${player.name} (${teamName}) ha marcado gol con el pase de ${asistidor.name} en el minuto ${formattedTime}`;
        } else {
          msg = `${player.name} (${teamName}) ha marcado gol en el minuto ${formattedTime}`;
          log = `‚öΩ **GOL INDIVIDUAL** | ${player.name} (${teamName}) ha marcado gol en el minuto ${formattedTime}`;
        }
      } else {
        msg = `${player.name} (${teamName}) ha marcado gol en el minuto ${formattedTime}`;
        log = `‚öΩ **GOL INDIVIDUAL** | ${player.name} (${teamName}) ha marcado gol en el minuto ${formattedTime}`;
      }
      room.sendAnnouncement(msg, null, 0x00FF00, "bold", 2); // Verde para goles
    }
  }

  sendDiscordLog(log); // Replicar el mensaje de gol en los logs
  touches.length = 0; // Limpiar los toques despu√©s de cada gol
};
